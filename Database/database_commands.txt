#---------------------------------------------------------------------------------------------------------------
Create database to store user's information
#---------------------------------------------------------------------------------------------------------------

create database user_info;

#---------------------------------------------------------------------------------------------------------------
Create Table to store user's login information
#---------------------------------------------------------------------------------------------------------------

create table user_info.user_login(user_id int(10) not null, 
                                    username varchar(45) primary key not null,
                                    Full_name varchar(45) not null,
                                    user_pass varchar(45) not null,
                                    email varchar(30) not null,
                                    contact int(12) not null,
                                    security_qns varchar(45) not null,
                                    security_qns_ans varchar(30) not null);

alter table user_info.user_login modify user_id int(10) not null auto_increment unique;

#---------------------------------------------------------------------------------------------------------------
Active Sessions Table
#---------------------------------------------------------------------------------------------------------------
create table active_sessions(session_id int(30) primary key not null,
                             username varchar(45) unique not null, 
                             date varchar(10), 
                             time varchar(10), 
                             otp varchar(10));

alter table active_sessions modify session_id int(30) not null auto_increment;
